<script>
document.getElementById('send-webhook-btn').addEventListener('click', function() {
    const url = document.getElementById('webhook-url').value;
    const eventType = document.getElementById('event-type').value;
    const statusDiv = document.getElementById('response-status');

    if (!url) {
        statusDiv.textContent = 'Error: Webhook URL is required.';
        statusDiv.style.color = 'red';
        return;
    }

    // Define mock payloads for different event types
    const mockPayloads = {
        'user.created': {
            event: 'user.created',
            userId: 'user-12345',
            email: 'test.user@example.com',
            timestamp: new Date().toISOString()
        },
        'api.subscribed': {
            event: 'api.subscribed',
            userId: 'user-12345',
            apiId: 'api-abcdef',
            plan: 'premium',
            timestamp: new Date().toISOString()
        },
        'key.revoked': {
            event: 'key.revoked',
            keyId: 'key-zyxw-9876',
            reason: 'compromised',
            timestamp: new Date().toISOString()
        }
    };

    const payload = mockPayloads[eventType];

    statusDiv.textContent = 'Sending event...';
    statusDiv.style.color = 'black';

    // Use the Fetch API to make the outbound HTTP POST request
    fetch(url, {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'X-Webhook-Simulator': 'Tyk-Portal'
        },
        body: JSON.stringify(payload)
    })
    .then(response => {
        statusDiv.textContent = `Success! Responded with status: ${response.status}`;
        statusDiv.style.color = 'green';
    })
    .catch(error => {
        statusDiv.textContent = `Failed to send webhook: ${error.message}`;
        statusDiv.style.color = 'red';
    });
});
</script>
